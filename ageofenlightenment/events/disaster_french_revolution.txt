#Declaration of Pillnitz
country_event = {
	id = french_revolution.7
	title = "french_revolution.7.t"
	desc = "french_revolution.7.d"
	picture = DIPLOMACY_eventPicture
	
	major = yes
	fire_only_once = yes
	
	trigger = {
		is_year = 1791
		OR = {
			is_month = 7
			is_month = 8
		}
		FRA = {
			has_disaster = french_revolution
			has_country_flag = flight_to_varennes
		}
		OR = {
			AND = {
				any_country = {
					is_part_of_hre = yes
				}
				is_emperor = yes
			}
			AND = {
				NOT = {
					any_country = {
						is_part_of_hre = yes
					}
				}
				capital_scope = {
					continent = europe
				}
				is_great_power = yes
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "french_revolution.7.a"
		FRA = {
			country_event = {
				id = french_revolution.11
				months = 8
			}
		}
		every_country = {
			limit = {
				capital_scope = {
					continent = europe
					NOT = {
						tag = RUS
						tag = FRA
						tag = DAN
						tag = SWE
					}
				}
			}
			country_event = {
				id = french_revolution.8
				days = 2
			}
		}
	}
}
