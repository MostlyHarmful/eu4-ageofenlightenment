country_event = {
	id = hidden.1
	
	title = "hidden.1.t"
	desc = "hidden.1.d"
	
	picture = SIEGE_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		OR = {
			culture_group = east_asian
			is_emperor_of_china = yes
		}
		any_owned_province = {
			chinese_region_trigger = yes
			NOT = { is_core = ROOT }
		}
	}
		
	option = {
		name = "hidden.1.a"
		every_owned_province = {
			limit = {
				chinese_region_trigger = yes
				NOT = { is_core = ROOT }
			}
			add_core = ROOT
		}
	}
}

		}

				NOT = { exists = KOR }
