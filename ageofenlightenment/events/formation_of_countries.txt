country_event = {
	id = formation_of_countries.2
	title = "formation_of_countries.2.t"
	desc = "formation_of_countries.2.d"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	major = yes
	
	trigger = {
		current_age = age_of_revolutions
		is_year = 1750
		NOT = { exists = BEL }
		is_free_or_tributary_trigger = yes
		num_of_cities = 6
		any_owned_province = {
			belgium_region_trigger = yes
			OR = {
				culture = flemish
				culture = wallonian
				is_core = BEL
			}
		}
		capital_scope = {
			NOT = { belgium_region_trigger = yes }
		}
		NOT = {
			OR = {
				primary_culture = flemish
				primary_culture = wallonian
			}
		}
		is_revolution_target = no
	}
	
	mean_time_to_happen = {
		months = 1000
		modifier = {
			factor = 0.05
			is_year = 1820
		}
		modifier = {
			factor = 0.75
			NOT = { accepted_culture = wallonian }
		}
		modifier = {
			factor = 0.75
			NOT = { accepted_culture = flemish }
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 2
				belgium_region_trigger = yes
				OR = {
					culture = wallonian
					culture = flemish
				}
			}
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 4
				belgium_region_trigger = yes
				OR = {
					culture = wallonian
					culture = flemish
				}
			}
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 6
				belgium_region_trigger = yes
				OR = {
					culture = wallonian
					culture = flemish
				}
			}
		}
		modifier = {
			factor = 0.75
			num_of_owned_provinces_with = {
				value = 8
				belgium_region_trigger = yes
				OR = {
					culture = wallonian
					culture = flemish
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					belgium_region_trigger = yes
					OR = {
						culture = flemish
						culture = wallonian
						is_core = BEL
					}
					is_capital = no
				}
				discover_country = BEL
				add_core = BEL
			}
			release = BEL
			BEL = {
				change_government = constitutional_monarchy
			}
			every_province = {
				limit = {
					belgium_region_trigger = yes
				}
				add_core = BEL
			}
		}
	}
	
	option = {
		name = "formation_of_countries.2.a"
		ROOT = {
			declare_war_with_cb = {
				who = BEL
				casus_belli = cb_insult
			}
		}
		hidden_effect = {
			BEL = {
				country_event = { id = formation_of_countries.3 days = 1 }
			}
			# Send messages to rivals allowing them to join
			every_enemy_country = {
				country_event = { id = formation_of_countries.4 days = 1 }
			}
		}
		ai_chance = { factor = 90 }
	}
	option = {
		name = "formation_of_countries.2.b"
		add_truce_with = BEL
		add_prestige = -10
		hidden_effect = {
			BEL = {
				country_event = { id = formation_of_countries.3 days = 1 }
			}
			# Send messages to rivals allowing them to join
			every_enemy_country = {
				country_event = { id = formation_of_countries.4 days = 10 }
			}
		}
		ai_chance = { factor = 10 }
	}
	option = {
		name = "formation_of_countries.2.c"
		ROOT = {
			declare_war_with_cb = {
				who = BEL
				casus_belli = cb_insult
			}
		}
		hidden_effect = {
			BEL = {
				country_event = { 
					id = formation_of_countries.3 
					days = 1 
				}
			}
			# Send messages to rivals allowing them to join
			every_enemy_country = {
				country_event = { 
					id = formation_of_countries.4 
					days = 10 
				}
			}
		}
		switch_tag = BEL
		ai_chance = { factor = 0 }
	}
}
