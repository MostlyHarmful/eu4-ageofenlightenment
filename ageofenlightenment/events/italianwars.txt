	desc = "italian_wars.2.d"

	fire_only_once = yes
	picture = BATTLE_eventPicture
	
	trigger = {
		tag = FRA
		is_year = 1480
		NOT = { is_year = 1550 }
		NOT = { senior_union_with = NAP }
		exists = NAP
		exists = MLO
		MLO = {
			is_part_of_hre = no
		}
		NAP = {
			is_part_of_hre = no
		}
		is_subject = no
	}
	
	mean_time_to_happen = {
		months = 30
	}
	
	option = {
		name = "italian_wars.2.a"
		declare_war_with_cb = {
		  who = NAP
		  casus_belli = cb_restore_personal_union
		}
		declare_war_with_cb = {
		  who = MLO
		  casus_belli = cb_restore_personal_union
		}
		add_casus_belli = {
		  target = NAP
		  type = cb_restore_personal_union
		  months = 36
		}
		add_casus_belli = {
		  target = MLO
		  type = cb_restore_personal_union
		  months = 36
		}
		add_prestige = 10
		ai_chance = { factor = 97 }
		hidden_effect = {
			every_enemy_country = {
				limit = {
					NOT = { tag = CAS }
					NOT = { tag = ARA }
					NOT = { tag = SPA }
				}
				country_event = { id = italian_wars.4 days = 1 }
			}
			CAS = {
				country_event = { 
					id = italian_wars.3 
					days = 300
				}
			}
			set_global_flag = italianwars
		}
	}
	option = {
		name = "italian_wars.2.b"
		add_prestige = -10
		ai_chance = { factor = 3 }
	}
}

			}
		}

			limit = {

					ai = yes

				add_casus_belli = {
				  target = NAP

				  months = 120
				}
			}

			limit = {
				MLO = {
					ai = yes

				}
