republicofchina = {
	potential = {
        NOT = {
                has_country_flag = had_republican_revolution
        }
		current_age = age_of_revolutions
		has_dlc = "Mandate of Heaven"
		is_emperor_of_china = yes
		num_of_owned_provinces_with = {
			value = 5
			OR = {
				region = xinan_region
				region = south_china_region
			}
		}
	}
	can_start = {
		has_any_disaster = no
		has_institution = enlightenment
		OR = {
			AND = {
				NOT = { culture_group = east_asian }
				NOT = { imperial_mandate = 75 }
				stability = -1
			}
			AND = { 
				culture_group = east_asian
				NOT = { imperial_mandate = 60 }
				stability = -2
			}
		}
	}
	can_stop = {
		OR = {
			AND = {
				imperial_mandate = 50
				stability = 1
			}
			has_any_disaster = yes
		}
	}
	
	modifier = {
		global_unrest = 5
		imperial_mandate = -0.1
	}
	progress = {
		modifier = {
			factor = 5
			NOT = { imperial_mandate = 25 }
		}
		modifier = {
			factor = 2.5
			NOT = { imperial_mandate = 50 }
		}
		modifier = {
			factor = 0.5
			NOT = { imperial_mandate = 75 }
		}
		modifier = {
			factor = 0.5
			absolutism = 25
			NOT = { absolutism = 50 }
		}
		modifier = {
			factor = 1
			absolutism = 50
			NOT = { absolutism = 75 }
		}
		modifier = {
			factor = 1.5
			absolutism = 75
		}
		modifier = {
			factor = 1
			NOT = { culture_group = east_asian }
		}
		modifier = {
			factor = -1
			full_idea_group = humanist_ideas
		}
	}
	can_end = {
		AND = {
			imperial_mandate = 100
			stability = 3
			NOT = { num_of_rebel_controlled_provinces = 1 }
		}
	}
	
	on_start = disaster_republicofchina.1
	on_end = disaster_republicofchina.2
	
	on_monthly = {
		events = {
			disaster_republicofchina.10
		}
		random_events = { 
			1000 = 0
			100 = disaster_republicofchina.3
			100 = disaster_republicofchina.4
		}
	}
}
